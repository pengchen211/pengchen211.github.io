1 简单动态字符串(simple dynamic string)

struct sdshdr {
   int len; // 字符串长度
   int free; // 空闲长度
   char buf[];
};

1.1 图示

1.2 优点

  1. 快速获取字符串长度
  2. 避免缓冲区溢出
  3. 减少修改字符串带来的内存重分配次数

    空间分配策略

    1. 空间预分配 对sds修改并需要扩展长度时，会按一定规则分配额外空间
      • 长度修改后将小于1MB，则分配与len同样大小的free
      • 将大于1MB，则分配1MB的free
    2. 惰性空间释放 用于优化sds字符串缩短操作,当字符串缩短时，不会立即回收多于的内存，会将其作为free
  4. 二进制安全
  5. 兼容C语言字符串函数

2 链表

typedef struct listNode {
    struct listNode *prev;
    struct listNode *next;
    void *value;
} listNode;

typedef struct list {
    listNode *head;
    listNode *tail;
    void *(*dup)(void *ptr);// 节点值复制函数
    void (*free)(void *ptr);// 节点值释放函数
    int (*match)(void *ptr, void *key);// 节点值对比函数
    unsigned long len;
} list;

3 字典

// 哈希表
typedef struct dictht {
		// 哈希表数组
    dictEntry **table;// dictEntry
		// 哈希表大小
    unsigned long size;
		// 哈希表大小掩码，用于计算索引值
		// 总是size - 1，决定一个key映射到table数组哪个索引上
    unsigned long sizemask;
		// 哈希表已有节点的数量
    unsigned long used;
} dictht;

// 哈希表节点
typedef struct dictEntry {
    void *key;
    union {
        void *val;
        uint64_t u64;
        int64_t s64;
        double d;
    } v;
    struct dictEntry *next;// 用来解决hash冲突
} dictEntry;

// 字典
typedef struct dict {
    // 类型特定函数
    dictType *type;
    // 私有函数
    void *privdata;
    // 哈希表
    // 一般只会使用ht[0]，ht[1]在rehash时使用
    dictht ht[2];
		// rehash索引
    long rehashidx; /* rehashing not in progress if rehashidx == -1 */
    unsigned long iterators; /* number of iterators currently running */
} dict;

3.2 rehash

  1. 集中式
  2. 渐进式
    1. 为ht[1]分配空间，字典同时持有2个hash表
    2. 字典中rehashindex置0
    3. 每次对字典crud，会同时将ht[0]上rehashindex索引上的所有键值对rehash到ht[1],并rehashindex++
    4. 完成后，rehashindex置-1
